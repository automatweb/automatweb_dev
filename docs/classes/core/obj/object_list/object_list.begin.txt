resets the internal iterator to the beginning of the list

<b>errors:</b>
none

<b>parameters:</b>
none

<b>returns:</b>
none

<b>examples:</b>
#php#
$ol = new object_list(array(
		"status" => STAT_ACTIVE
	));
for ($o = $ol->begin(); !$ol->end(); $o =& $ol->next();)
{
	$o->set_name("kala");
	$o->save();
}#/php#
